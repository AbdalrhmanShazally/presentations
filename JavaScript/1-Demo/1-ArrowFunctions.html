        // ES6
        const x2 = (x, y) => x * y;

');

        //Regular Function:
        var hello1 = function() {
            document.getElementById("demoStandard").innerHTML += this;
        }
        //Arrow Function:
        var hello2 = () => {
            document.getElementById("demoArrow").innerHTML += this;
        }

//The window object calls the function:
        window.addEventListener("load", hello1);
        window.addEventListener("load", hello2);

//A button object calls the function:
        document.getElementById("btnStandard").addEventListener("click", hello1);
        document.getElementById("btnArrow").addEventListener("click", hello2);
    })();
